
const express = require('express');
const path = require('path');
const cors = require('cors');
const bodyParser = require('body-parser');

// Import modules
const database = require('./src/database');
const errorHandler = require('./src/errorHandler');
const { router: authRoutes } = require('./src/auth');
const gameRoutes = require('./src/game');
const adminRoutes = require('./src/admin');
const mediaRoutes = require('./src/media');
const characterRoutes = require('./src/characters');
const chatRoutes = require('./src/chat');

const app = express();
const PORT = process.env.PORT || 5000;

// Middleware
app.use(cors());
app.use(bodyParser.json());
app.use(bodyParser.urlencoded({ extended: true }));
app.use(express.static(path.join(__dirname, 'public')));

// Routes
app.use('/api/auth', authRoutes);
app.use('/api/game', gameRoutes);
app.use('/api/admin', adminRoutes);
app.use('/api/media', mediaRoutes);
app.use('/api/characters', characterRoutes);
app.use('/api/chat', chatRoutes);

// Serve uploaded files
app.use('/uploads', express.static(path.join(__dirname, 'uploads')));

// Global error handler (must be after routes)
app.use(errorHandler.middleware);

// Serve main game page
app.get('/', (req, res) => {
  res.sendFile(path.join(__dirname, 'public', 'index.html'));
});

// Initialize database and start server
database.init().then(() => {
  app.listen(PORT, '0.0.0.0', () => {
    console.log(`ClassikLust Game Server running on port ${PORT}`);
  });
}).catch(err => {
  console.error('Failed to initialize database:', err);
  process.exit(1);
});

// Global error handling
process.on('uncaughtException', errorHandler.logError);
process.on('unhandledRejection', errorHandler.logError);
